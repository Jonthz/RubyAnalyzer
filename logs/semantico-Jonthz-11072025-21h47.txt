  def initialize(name)
    @name = name
  end

  def add_numbers(num1, num2)
    result = num1.to_i + num2.to_i

    puts "Suma realizada"
    puts result
    return result
  end

  def multiply_values(val1, val2)
    product = val1.to_i * val2.to_i
    puts "Multiplicación completada"
    puts product
    return product
  end

  def get_memory
    puts "Memoria actual"
    puts @name
    return @name
  end
end
