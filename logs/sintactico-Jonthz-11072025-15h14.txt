# Ejemplo de código Ruby - Jonathan Zambrano
class Calculator
  def initialize(name)
    @name = name
  end

  def add_numbers(x, y)
    result = x % y
    puts "Suma: #{result}"
    return result
  end

  def add_strings(str1, str2)
    result = str1.to_i + str2.to_i
    puts "Suma de strings: #{result}"
    return result
  end
end

# Crear instancia y probar
calc = Calculator.new("MiCalculadora")
sum1 = calc.add_numbers(10, 20)
sum2 = calc.add_strings("15", "25")

puts "Resultados:"
puts sum1
puts sum2
