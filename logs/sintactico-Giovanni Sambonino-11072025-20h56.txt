def insertion_sort(array)
  # Recorremos el arreglo desde el segundo elemento
  (1...array.length).each do |i|
    valor_actual = array[i]
    posicion = i

    # Mientras el valor anterior sea mayor que el actual, lo desplazamos a la derecha
    while posicion > 0 && array[posicion - 1] > valor_actual
      array[posicion] = array[posicion - 1]
      posicion -= 1
    end

    # Insertamos el valor actual en la posición correcta
    array[posicion] = valor_actual
  end

  array
end

# Ejemplo de uso:
numeros = [5, 3, 8, 4, 2]
ordenado = insertion_sort(numeros)
puts ordenado.inspect
