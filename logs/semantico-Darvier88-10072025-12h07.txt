def quick_sort(arr)
  if arr.length <= 1
    return arr
  end

  pivot_index = 0
  pivot = arr[pivot_index]
  left = []
  right = []

  i = 0
  while i < arr.length
    if i != pivot_index
      if arr[i] < pivot
        left = left + [arr[i]]
      else
        right = right + [arr[i]]
      end
    end
    i = i + 1
  end
  sorted_left = quick_sort(left)
  sorted_right = quick_sort(right)
  return sorted_left + [pivot] + sorted_right
end
