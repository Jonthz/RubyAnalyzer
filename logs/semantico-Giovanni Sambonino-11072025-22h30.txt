# Algoritmo de Ordenamiento por Inserción - Versión Ultra Conservadora
def insertion_sort(arr)
  if arr.length == 0
    return arr
  end

  for i in 1..(arr.length - 1)
    key = arr[i]
    j = i - 1

    while j >= 0 && arr[j] > key
      arr[j + 1] = arr[j]
      j = j - 1
    end
    arr[j + 1] = key
  end
  return arr
end

def verificar_ordenamiento(arr)
  for i in 1..(arr.length - 1)
    if arr[i - 1] > arr[i]
      return false
    end
  end
  return true
end

def sum(a, b)
  resultado = a + b
  return resultado
end
