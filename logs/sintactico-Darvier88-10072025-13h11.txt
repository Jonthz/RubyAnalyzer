def suma_mayores(arr, limite)
  suma = 0
  for i in 0..(arr.length - 1)
    if arr[i] > limite
      suma = suma + arr[i]
    end
  end
  return suma
end

numeros = [1, 5, 10, 15, 20]
resultado = suma_mayores(numeros, 10)
puts resultado
