class Animal
  def initialize(name)
    @name = name
  end

  def speak()
    puts "El animal hace un sonido."
  end

  def info()
    puts @name
  end
end

class Dog < Animal
  def initialize(name, breed)
    super(name)
    @breed = breed
  end

  

  def info()
    puts @name
    puts @breed
  end
end

class Cat < Animal
  def initialize(name)
    super(name)
  end



  def info()
    puts @name
  end
end

dog = Dog.new("Rex", "Pastor Aleman")
cat = Cat.new("Whiskers")

dog.info()
cat.info()
