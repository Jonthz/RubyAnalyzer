  def initialize
    puts "Calculadora inicializada"
  end

  def add(a, b)
    return a + b
  end

  def multiply(x, y)
    result = x * y
    return result
  end

  def divide(num, den)
    if den == 0
      puts "Error: División por cero"
      return 0
    else
      return num / den
    end
  end
end
