def insertion_sort(arr)
  # Verificar si el array está vacío
  if arr.length == 0
    return arr
  end

  for i in 1..(arr.length - 1)
    key = arr[i]
    j = i - 1

    # Mover elementos mayores hacia adelante
    while j >= 0 && arr[j] > key
      arr[j + 1] = arr[j]
      j -= 1
    end
    arr[j + 1] = key
  end
  return arr
end

def verificar_ordenamiento(arr)
  for i in 1..(arr.length - 1)
    if arr[i - 1] > arr[i]
      return false
    end
  end
  return true
end
