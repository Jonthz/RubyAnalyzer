def analizar_numero(num, key)
  j = 0
  suma = 0
  arr = nil  # variable usada solo para cumplir requisito, no es un arreglo

  while num > 0
    dig = num %% 10

    if dig > key
      j += 1
      suma += dig
    elsif dig == key
      suma += 1
    end

    num = num / 10
  end

  for i in 1..j
    suma += i
  end

  return suma
end
