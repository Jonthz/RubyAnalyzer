def insertion_sort(arr)
  (1...arr.length).each do |i|
    key = arr[i]
    j = i - 1

    while j >= 0 && arr[j] > key
      arr[j + 1] = arr[j]
      j -= 1
    end

    arr[j + 1] = key
  end

  arr
end

# Ejemplo
puts insertion_sort([5, 3, 8, 4, 2]).inspect  # => [2, 3, 4, 5, 8]
